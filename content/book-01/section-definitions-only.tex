\MatterChapter{Definitions}

\MatterSubsection{subsec:foundations:definitions:motion}{Motion}{Hamza Bin Aamir}
\begin{Definition}
A debate prompt setting the policy change or claim to be argued.
\end{Definition}

\MatterSubsection{subsec:foundations:definitions:case}{Case}{Hamza Bin Aamir}
\begin{Definition}
A structured set of arguments, examples, and framing used to support a side of the motion.
\end{Definition}
