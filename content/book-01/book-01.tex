\MatterSubsection{subsec:foundations:workflow}{Replacement workflow}{Hamza Bin Aamir}

\begin{Factsheet}
This matter file is designed for binder use. Each \textbf{subsection} (the level created via \texttt{\string\MatterSubsection}) starts on a new page so you can reprint only the pages you changed.
\end{Factsheet}

\begin{Checklist}
  \item Keep each subsection self-contained.
  \item Put sources and caveats inside the same subsection.
  \item When updating, only reprint the affected subsection pages.
\end{Checklist}

\MatterSubsection{subsec:foundations:style}{Man-page style}{Hamza Bin Aamir}

\begin{FirstPrinciples}
Use compact headings, short paragraphs, and structured blocks. Prioritize scanability over prose.
\end{FirstPrinciples}
